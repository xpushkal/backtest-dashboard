```rust
use chrono::{NaiveDate, Weekday, Datelike};
use serde::Deserialize;
use std::collections::HashMap;

#[derive(Debug, Deserialize, Clone)]
pub struct TransitionRule {
    pub from: NaiveDate,
    pub to: NaiveDate,
    #[serde(rename = "type")]
    pub expiry_type: String,  // "weekly" or "monthly"
    pub day: String,          // "Wednesday", "Thursday", "Friday", "last_thursday", "last_friday"
}

#[derive(Debug, Clone)]
pub struct ExpiryCalendar {
    rules: HashMap<String, Vec<TransitionRule>>,
}

impl ExpiryCalendar {
    /// Load from config/expiry_calendar.toml
    pub fn from_toml(path: &str) -> Result<Self, Box<dyn std::error::Error>> { ... }

    /// Resolve the next expiry date for the given symbol and date.
    /// Returns (expiry_date: NaiveDate, expiry_type: &str, dte: i64)
    pub fn next_expiry(&self, symbol: &str, date: NaiveDate) -> Option<(NaiveDate, String, i64)> { ... }

    /// Get the expiry type (weekly/monthly) for a symbol on a given date
    pub fn expiry_type(&self, symbol: &str, date: NaiveDate) -> Option<String> { ... }
}
```

Algorithm for `next_expiry`:
1. Find the applicable TransitionRule for (symbol, date)
2. If `expiry_type == "weekly"`:
   - Parse the day name to `Weekday`
   - Find the next occurrence of that weekday from `date` (inclusive)
   - If `date` is after market close on the expiry day, advance to next week
3. If `expiry_type == "monthly"`:
   - Parse "last_thursday" or "last_friday"
   - Find the last occurrence of that weekday in the current month
   - If `date` is after that date, find the last occurrence in the next month
4. Return (expiry_date, expiry_type, expiry_date - date as dte)

Helper functions:
- `fn next_weekday(date: NaiveDate, day: Weekday) -> NaiveDate`
- `fn last_weekday_of_month(year: i32, month: u32, day: Weekday) -> NaiveDate`
- `fn parse_day(s: &str) -> Weekday` — handles both "Wednesday" and "last_thursday"

use quantedge_data::ExpiryCalendar;
use chrono::NaiveDate;

#[test]
fn test_banknifty_weekly_expiry() {
    let cal = ExpiryCalendar::from_toml("../../config/expiry_calendar.toml").unwrap();
    // Monday 2024-01-08 → next Wednesday = 2024-01-10
    let (expiry, etype, dte) = cal.next_expiry("BANKNIFTY", NaiveDate::from_ymd_opt(2024, 1, 8).unwrap()).unwrap();
    assert_eq!(etype, "weekly");
    assert_eq!(expiry.weekday(), chrono::Weekday::Wed);
    assert!(dte >= 0);
}

#[test]
fn test_banknifty_monthly_after_transition() {
    let cal = ExpiryCalendar::from_toml("../../config/expiry_calendar.toml").unwrap();
    // 2024-11-05 → should be monthly, last Thursday
    let (expiry, etype, _) = cal.next_expiry("BANKNIFTY", NaiveDate::from_ymd_opt(2024, 11, 5).unwrap()).unwrap();
    assert_eq!(etype, "monthly");
    assert_eq!(expiry.weekday(), chrono::Weekday::Thu);
}

#[test]
fn test_nifty_weekly_before_transition() {
    let cal = ExpiryCalendar::from_toml("../../config/expiry_calendar.toml").unwrap();
