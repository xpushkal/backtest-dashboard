Natural cubic spline algorithm (standard Thomas algorithm):
1. Given n+1 data points (x_0, y_0), ..., (x_n, y_n)
2. Compute h_i = x_{i+1} - x_i for each interval
3. Set up tridiagonal system for second derivatives (natural BC: S''(x_0) = S''(x_n) = 0)
4. Solve via Thomas algorithm
5. Compute a_i, b_i, c_i, d_i coefficients for each interval
6. For interpolation at x: binary search for interval, evaluate a + b*(x-x_i) + c*(x-x_i)^2 + d*(x-x_i)^3

use quantedge_data::IvSurface;

#[test]
fn test_interpolation_at_data_points() {
    // Spline should pass through data points exactly
    let points = vec![(-5.0, 0.20), (0.0, 0.18), (5.0, 0.22), (10.0, 0.25)];
    let surface = IvSurface::from_points(&points).unwrap();
    assert!((surface.interpolate(0.0) - 0.18).abs() < 1e-10);
    assert!((surface.interpolate(-5.0) - 0.20).abs() < 1e-10);
    assert!((surface.interpolate(10.0) - 0.25).abs() < 1e-10);
}

#[test]
fn test_interpolation_between_points() {
    let points = vec![(-5.0, 0.20), (0.0, 0.18), (5.0, 0.22)];
    let surface = IvSurface::from_points(&points).unwrap();
    let iv_at_2 = surface.interpolate(2.0);
    // Should be between min and max IV (reasonable interpolation)
    assert!(iv_at_2 > 0.15 && iv_at_2 < 0.30);
}

#[test]
fn test_extrapolation_flat() {
    let points = vec![(-5.0, 0.20), (0.0, 0.18), (5.0, 0.22)];
    let surface = IvSurface::from_points(&points).unwrap();
    // Outside range should return boundary value
    assert!((surface.interpolate(-10.0) - 0.20).abs() < 1e-10);
    assert!((surface.interpolate(15.0) - 0.22).abs() < 1e-10);
}

#[test]
fn test_too_few_points() {
    let points = vec![(0.0, 0.18), (5.0, 0.22)];
    assert!(IvSurface::from_points(&points).is_none());
}

#[test]
fn test_monotonic_strike_offsets() {
    // Unsorted input should still work
    let points = vec![(5.0, 0.22), (-5.0, 0.20), (0.0, 0.18)];
    let surface = IvSurface::from_points(&points).unwrap();
    assert!((surface.interpolate(0.0) - 0.18).abs() < 1e-10);
}

use criterion::{criterion_group, criterion_main, Criterion};
use quantedge_data::{IvSurface};

fn bench_iv_interpolation(c: &mut Criterion) {
    let points: Vec<(f64, f64)> = (-10..=10)
        .map(|i| (i as f64, 0.18 + (i as f64).abs() * 0.005))
        .collect();
    let surface = IvSurface::from_points(&points).unwrap();
    
    c.bench_function("iv_interpolation", |b| {
        b.iter(|| surface.interpolate(2.5))
    });
}

fn bench_iv_build(c: &mut Criterion) {
    let points: Vec<(f64, f64)> = (-10..=10)
        .map(|i| (i as f64, 0.18 + (i as f64).abs() * 0.005))
        .collect();
    
    c.bench_function("iv_surface_build", |b| {
        b.iter(|| IvSurface::from_points(&points))
    });
}

criterion_group!(benches, bench_iv_interpolation, bench_iv_build);
