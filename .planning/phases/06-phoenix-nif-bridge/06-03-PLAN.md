# Plan 06-03: Postgres Ecto Schemas and Migrations

## Objective
Create Ecto schemas, migrations, and context modules for strategies, backtest_runs, and optimizer_runs tables in Postgres. These tables store transactional metadata — the actual trade/equity data goes to DuckDB (Plan 06-04).

## Requirements Covered
- STR-01: Postgres tables: strategies, backtest_runs, optimizer_runs

## Context
- PRD defines exact schemas (see P6 — Postgres Tables section)
- UUIDs for all primary keys via `gen_random_uuid()`
- `result_summary` stored as JSONB (top-level metrics only)
- Status enum: pending → running → completed → failed
- Ecto 3.11+ with `postgrex` adapter

## Implementation Steps

### Step 1: Create strategies migration + schema
```elixir
# Migration
defmodule QuantEdge.Repo.Migrations.CreateStrategies do
  use Ecto.Migration

  def change do
    create table(:strategies, primary_key: false) do
      add :id, :binary_id, primary_key: true
      add :name, :string, null: false
      add :underlying, :string, null: false
      add :config_toml, :text, null: false
      timestamps(type: :utc_datetime)
    end

    create index(:strategies, [:underlying])
    create unique_index(:strategies, [:name])
  end
end

# Schema
defmodule QuantEdge.Strategies.Strategy do
  use Ecto.Schema
  import Ecto.Changeset

  @primary_key {:id, :binary_id, autogenerate: true}
  schema "strategies" do
    field :name, :string
    field :underlying, :string
    field :config_toml, :string
    has_many :backtest_runs, QuantEdge.Runs.BacktestRun
    timestamps(type: :utc_datetime)
  end

  def changeset(strategy, attrs) do
    strategy
    |> cast(attrs, [:name, :underlying, :config_toml])
    |> validate_required([:name, :underlying, :config_toml])
    |> unique_constraint(:name)
  end
end
```

### Step 2: Create backtest_runs migration + schema
```elixir
# Migration
defmodule QuantEdge.Repo.Migrations.CreateBacktestRuns do
  use Ecto.Migration

  def change do
    create table(:backtest_runs, primary_key: false) do
      add :id, :binary_id, primary_key: true
      add :strategy_id, references(:strategies, type: :binary_id), null: false
      add :status, :string, null: false, default: "pending"
      add :date_from, :date, null: false
      add :date_to, :date, null: false
      add :capital, :decimal, null: false
      add :started_at, :utc_datetime
      add :completed_at, :utc_datetime
      add :result_summary, :map  # JSONB
      timestamps(type: :utc_datetime)
    end

    create index(:backtest_runs, [:strategy_id])
    create index(:backtest_runs, [:status])
  end
end

# Schema
defmodule QuantEdge.Runs.BacktestRun do
  use Ecto.Schema
  import Ecto.Changeset

  @primary_key {:id, :binary_id, autogenerate: true}
  schema "backtest_runs" do
    field :status, :string, default: "pending"
    field :date_from, :date
    field :date_to, :date
    field :capital, :decimal
    field :started_at, :utc_datetime
    field :completed_at, :utc_datetime
    field :result_summary, :map
    belongs_to :strategy, QuantEdge.Strategies.Strategy, type: :binary_id
    timestamps(type: :utc_datetime)
  end

  def changeset(run, attrs) do
    run
    |> cast(attrs, [:strategy_id, :status, :date_from, :date_to, :capital,
                     :started_at, :completed_at, :result_summary])
    |> validate_required([:strategy_id, :date_from, :date_to, :capital])
    |> validate_inclusion(:status, ~w(pending running completed failed))
  end
end
```

### Step 3: Create optimizer_runs migration + schema

### Step 4: Create context modules
```elixir
# lib/quantedge/strategies.ex
defmodule QuantEdge.Strategies do
  import Ecto.Query
  alias QuantEdge.Repo
  alias QuantEdge.Strategies.Strategy

  def list_strategies, do: Repo.all(Strategy)
  def get_strategy!(id), do: Repo.get!(Strategy, id)
  def create_strategy(attrs), do: %Strategy{} |> Strategy.changeset(attrs) |> Repo.insert()
  def update_strategy(%Strategy{} = s, attrs), do: s |> Strategy.changeset(attrs) |> Repo.update()
  def delete_strategy(%Strategy{} = s), do: Repo.delete(s)
end

# lib/quantedge/runs.ex
defmodule QuantEdge.Runs do
  import Ecto.Query
  alias QuantEdge.Repo
  alias QuantEdge.Runs.BacktestRun

  def create_run(attrs), do: %BacktestRun{} |> BacktestRun.changeset(attrs) |> Repo.insert()
  def get_run!(id), do: Repo.get!(BacktestRun, id) |> Repo.preload(:strategy)
  def update_status(id, status) do
    get_run!(id) |> BacktestRun.changeset(%{status: to_string(status)}) |> Repo.update()
  end
  def store_result(id, result_map) do
    get_run!(id) |> BacktestRun.changeset(%{
      status: "completed",
      completed_at: DateTime.utc_now(),
      result_summary: result_map
    }) |> Repo.update()
  end
end
```
