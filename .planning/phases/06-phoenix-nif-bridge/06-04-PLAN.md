# Plan 06-04: DuckDB Schema, GenServer Writer, Batch Insert Helpers

## Objective
Create DuckDB tables for analytical data (trades, equity_curves, metrics, optimizer_results) and a serialized writer GenServer that prevents concurrent write conflicts. Batch insert helpers convert NIF output into DuckDB rows.

## Requirements Covered
- STR-02: DuckDB tables: trades, equity_curves, metrics, optimizer_results
- STR-04: Single GenServer writer for DuckDB (no concurrent write conflicts)

## Context
- DuckDB is embedded (no separate server) via `duckdbex` hex package
- DuckDB is columnar — optimized for analytical queries (monthly heatmaps, metric aggregation)
- All writes go through a single GenServer to prevent WAL conflicts
- PRD specifies exact table schemas (see P6 — DuckDB Tables)
- DB file location: `priv/duckdb/quantedge.duckdb`

## Implementation Steps

### Step 1: Create DuckDB GenServer writer
```elixir
# lib/quantedge/duck/writer.ex
defmodule QuantEdge.Duck.Writer do
  use GenServer

  @db_path "priv/duckdb/quantedge.duckdb"

  def start_link(opts \\ []) do
    GenServer.start_link(__MODULE__, opts, name: __MODULE__)
  end

  def init(_opts) do
    {:ok, db} = Duckdbex.open(@db_path)
    {:ok, conn} = Duckdbex.connection(db)
    create_tables(conn)
    {:ok, %{db: db, conn: conn}}
  end

  # Public API — all writes serialized through GenServer
  def insert_trades(run_id, trades),          do: GenServer.call(__MODULE__, {:insert_trades, run_id, trades})
  def insert_equity_curve(run_id, curve),     do: GenServer.call(__MODULE__, {:insert_equity, run_id, curve})
  def insert_metrics(run_id, metrics_map),    do: GenServer.call(__MODULE__, {:insert_metrics, run_id, metrics_map})
  def insert_optimizer_results(run_id, results), do: GenServer.call(__MODULE__, {:insert_optimizer, run_id, results})

  # Reads go directly (DuckDB supports concurrent reads)
  def query(sql), do: GenServer.call(__MODULE__, {:query, sql})
end
```

### Step 2: Create DuckDB tables
```sql
-- trades (all legs of all runs — high row count, columnar queries)
CREATE TABLE IF NOT EXISTS trades (
  run_id        VARCHAR,
  trade_id      INTEGER,
  entry_time    TIMESTAMP,
  exit_time     TIMESTAMP,
  exit_reason   VARCHAR,
  legs          JSON,
  pnl_gross     DOUBLE,
  pnl_net       DOUBLE,
  hold_bars     INTEGER,
  greeks_entry  JSON,
  greeks_exit   JSON
);

-- equity_curves
CREATE TABLE IF NOT EXISTS equity_curves (
  run_id        VARCHAR,
  date          DATE,
  equity        DOUBLE,
  drawdown_pct  DOUBLE
);

-- metrics (one row per metric per run — fast aggregation)
CREATE TABLE IF NOT EXISTS metrics (
  run_id        VARCHAR,
  metric_name   VARCHAR,
  metric_value  DOUBLE
);

-- optimizer_results
CREATE TABLE IF NOT EXISTS optimizer_results (
  optimizer_run_id VARCHAR,
  combo_index      INTEGER,
  params           JSON,
  metrics          JSON
);
```

### Step 3: Implement batch insert handlers
```elixir
def handle_call({:insert_trades, run_id, trades}, _from, state) do
  stmt = Duckdbex.prepare(state.conn,
    "INSERT INTO trades VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)")
  for {trade, idx} <- Enum.with_index(trades) do
    Duckdbex.execute(stmt, [
      run_id, idx,
      trade["entry_time"], trade["exit_time"],
      trade["exit_reason"], Jason.encode!(trade["legs"]),
      trade["pnl_gross"], trade["pnl_net"],
      trade["hold_bars"],
      Jason.encode!(trade["greeks_entry"] || %{}),
      Jason.encode!(trade["greeks_exit"] || %{})
    ])
  end
  {:reply, :ok, state}
end

def handle_call({:insert_equity, run_id, curve}, _from, state) do
  stmt = Duckdbex.prepare(state.conn,
    "INSERT INTO equity_curves VALUES (?, ?, ?, ?)")
  for point <- curve do
    Duckdbex.execute(stmt, [run_id, point["date"], point["equity"], point["drawdown_pct"]])
  end
  {:reply, :ok, state}
end

def handle_call({:insert_metrics, run_id, metrics_map}, _from, state) do
  stmt = Duckdbex.prepare(state.conn,
    "INSERT INTO metrics VALUES (?, ?, ?)")
  for {name, value} <- metrics_map do
    Duckdbex.execute(stmt, [run_id, to_string(name), value])
  end
  {:reply, :ok, state}
end
```

### Step 5: Add DuckDB reader module for common queries
```elixir
# lib/quantedge/duck/reader.ex
defmodule QuantEdge.Duck.Reader do
  alias QuantEdge.Duck.Writer

  def get_trades(run_id) do
    Writer.query("SELECT * FROM trades WHERE run_id = '#{run_id}' ORDER BY trade_id")
  end

  def get_equity_curve(run_id) do
    Writer.query("SELECT * FROM equity_curves WHERE run_id = '#{run_id}' ORDER BY date")
  end

  def get_metrics(run_id) do
    Writer.query("SELECT metric_name, metric_value FROM metrics WHERE run_id = '#{run_id}'")
  end

  def get_monthly_pnl(run_id) do
    Writer.query("""
    SELECT EXTRACT(YEAR FROM exit_time) AS year,
           EXTRACT(MONTH FROM exit_time) AS month,
           SUM(pnl_net) AS pnl
    FROM trades
    WHERE run_id = '#{run_id}'
    GROUP BY year, month
    ORDER BY year, month
    """)
  end
end
```
