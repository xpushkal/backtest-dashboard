```elixir
# test/quantedge/nif_test.exs
defmodule QuantEdge.NIFTest do
  use ExUnit.Case, async: true

  describe "run_backtest/2" do
    test "returns {:ok, json} for valid strategy" do
      strategy_toml = File.read!("config/strategies/example_short_call.toml")
      opts = Jason.encode!(%{
        symbol: "BANKNIFTY",
        date_from: "2024-01-01",
        date_to: "2024-01-31",
        capital: 500_000.0,
        lot_size: 15,
        data_dir: "Data/parquet"
      })
      assert {:ok, json} = QuantEdge.NIF.run_backtest(strategy_toml, opts)
      result = Jason.decode!(json)
      assert is_list(result["trades"])
      assert is_map(result["metrics"])
    end

    test "returns {:error, reason} for invalid strategy JSON" do
      assert {:error, _reason} = QuantEdge.NIF.run_backtest("not valid", "{}")
    end
  end

  describe "NIF overhead" do
    test "empty call round-trip < 5ms" do
      # Benchmark NIF call overhead
      {time_us, _result} = :timer.tc(fn ->
        for _ <- 1..100 do
          try do
            QuantEdge.NIF.run_backtest("", "{}")
          rescue
            _ -> :ok
          end
        end
      end)
      avg_ms = time_us / 100 / 1_000
      assert avg_ms < 5.0, "NIF overhead #{avg_ms}ms exceeds 5ms target"
    end
  end
end
```

### Step 4: DuckDB writer serialization test
```elixir
# test/quantedge/duck/writer_test.exs
defmodule QuantEdge.Duck.WriterTest do
  use ExUnit.Case

  alias QuantEdge.Duck.{Writer, Reader}

  test "concurrent writes don't crash" do
    run_id = Ecto.UUID.generate()
    trades = [%{
      "entry_time" => "2024-01-15 09:30:00",
      "exit_time" => "2024-01-15 10:00:00",
      "exit_reason" => "time_exit",
      "legs" => [],
      "pnl_gross" => 1000.0,
      "pnl_net" => 900.0,
      "hold_bars" => 30,
      "greeks_entry" => nil,
      "greeks_exit" => nil
    }]

    # Spawn 5 concurrent writes
    tasks = for _ <- 1..5 do
      Task.async(fn -> Writer.insert_trades(run_id, trades) end)
    end
    results = Task.await_many(tasks, 5_000)
    assert Enum.all?(results, &(&1 == :ok))
  end

  test "inserted trades are queryable" do
    run_id = Ecto.UUID.generate()
    Writer.insert_trades(run_id, [%{
      "entry_time" => "2024-01-15 09:30:00",
      "exit_time" => "2024-01-15 10:00:00",
      "exit_reason" => "stop_loss",
      "legs" => [],
      "pnl_gross" => -500.0,
      "pnl_net" => -600.0,
      "hold_bars" => 15,
      "greeks_entry" => nil,
      "greeks_exit" => nil
    }])

    {:ok, result} = Reader.get_trades(run_id)
    assert length(result) >= 1
  end
end
```
