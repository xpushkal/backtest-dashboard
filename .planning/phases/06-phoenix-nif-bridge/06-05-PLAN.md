# Plan 06-05: Oban Workers with PubSub Integration

## Objective
Implement Oban background workers (BacktestWorker, OptimizerWorker, PortfolioWorker) that execute NIF calls, store results in both Postgres (summary) and DuckDB (detailed), and broadcast real-time progress/completion events via Phoenix.PubSub.

## Requirements Covered
- STR-03: Oban job queue: BacktestWorker, OptimizerWorker, PortfolioWorker with PubSub
- NIF-05: PubSub progress broadcasting during long-running NIF operations

## Context
- Oban 2.17+ with Postgres-backed queue
- Workers execute on `max_attempts: 1` (backtests are not idempotent-safe to retry blindly)
- PubSub topic pattern: `"run:{run_id}"` for backtest, `"optimizer:{run_id}"` for optimizer
- PRD defines exact worker pattern (see P6 — Oban Workers section)
- Queue names: `:backtests`, `:optimizers`, `:portfolios`

### Step 2: Implement BacktestWorker
```elixir
# lib/quantedge/workers/backtest_worker.ex
defmodule QuantEdge.Workers.BacktestWorker do
  use Oban.Worker, queue: :backtests, max_attempts: 1

  alias QuantEdge.{NIF, Runs}
  alias QuantEdge.Duck.Writer

  @impl Oban.Worker
  def perform(%Oban.Job{args: %{"run_id" => run_id}}) do
    run = Runs.get_run!(run_id)
    strategy = run.strategy

    # 1. Broadcast: running
    Runs.update_status(run_id, :running)
    broadcast(run_id, {:status, :running})

    # 2. Build NIF inputs
    opts = Jason.encode!(%{
      symbol: strategy.underlying,
      date_from: Date.to_iso8601(run.date_from),
      date_to: Date.to_iso8601(run.date_to),
      capital: Decimal.to_float(run.capital),
      lot_size: get_lot_size(strategy.underlying),
      data_dir: Application.get_env(:quantedge, :data_dir)
    })

    # 3. Execute NIF
    case NIF.run_backtest(strategy.config_toml, opts) do
      {:ok, result_json} ->
        result = Jason.decode!(result_json)

        # 4. Store in DuckDB (detailed)
        Writer.insert_trades(run_id, result["trades"])
        Writer.insert_equity_curve(run_id, result["equity_curve"])
        Writer.insert_metrics(run_id, result["metrics"])

        # 5. Store summary in Postgres
        Runs.store_result(run_id, extract_summary(result["metrics"]))

        # 6. Broadcast: completed
        broadcast(run_id, {:completed, result["metrics"]})
        :ok

      {:error, reason} ->
        Runs.update_status(run_id, :failed)
        broadcast(run_id, {:error, reason})
        {:error, reason}
    end
  end

  defp broadcast(run_id, message) do
    Phoenix.PubSub.broadcast(QuantEdge.PubSub, "run:#{run_id}", message)
  end

  defp extract_summary(metrics) do
    Map.take(metrics, ~w(total_pnl_net cagr win_rate_pct max_drawdown_pct
                          sharpe_ratio profit_factor total_trades))
  end

  defp get_lot_size("BANKNIFTY"), do: 15
  defp get_lot_size("NIFTY"), do: 25
  defp get_lot_size("SENSEX"), do: 10
  defp get_lot_size(_), do: 1
end
```

### Step 3: Implement OptimizerWorker
```elixir
# lib/quantedge/workers/optimizer_worker.ex
defmodule QuantEdge.Workers.OptimizerWorker do
  use Oban.Worker, queue: :optimizers, max_attempts: 1

  alias QuantEdge.{NIF, Runs}
  alias QuantEdge.Duck.Writer

  @impl Oban.Worker
  def perform(%Oban.Job{args: %{"optimizer_run_id" => run_id}}) do
    run = Runs.get_optimizer_run!(run_id)

    broadcast(run_id, {:status, :running})

    case NIF.run_optimizer(
      run.strategy.config_toml,
      Jason.encode!(run.param_grid)
    ) do
      {:ok, results_json} ->
        results = Jason.decode!(results_json)
        Writer.insert_optimizer_results(run_id, results)
        Runs.update_optimizer_status(run_id, :completed, length(results))
        broadcast(run_id, {:completed, length(results)})
        :ok

      {:error, reason} ->
        Runs.update_optimizer_status(run_id, :failed, 0)
        broadcast(run_id, {:error, reason})
        {:error, reason}
    end
  end

  defp broadcast(run_id, message) do
    Phoenix.PubSub.broadcast(QuantEdge.PubSub, "optimizer:#{run_id}", message)
  end
end
```

### Step 4: Implement PortfolioWorker
```elixir
# lib/quantedge/workers/portfolio_worker.ex
defmodule QuantEdge.Workers.PortfolioWorker do
  use Oban.Worker, queue: :portfolios, max_attempts: 1

  @impl Oban.Worker
  def perform(%Oban.Job{args: %{"portfolio_run_id" => run_id, "strategies" => strategies_json}}) do
    broadcast(run_id, {:status, :running})

    case QuantEdge.NIF.run_portfolio(strategies_json, "{}") do
      {:ok, result_json} ->
        result = Jason.decode!(result_json)
        # Store per-strategy results + combined portfolio metrics
        broadcast(run_id, {:completed, result})
        :ok

      {:error, reason} ->
        broadcast(run_id, {:error, reason})
        {:error, reason}
    end
  end

  defp broadcast(run_id, message) do
    Phoenix.PubSub.broadcast(QuantEdge.PubSub, "portfolio:#{run_id}", message)
  end
end
```
