```rust
use clap::Parser;
use quantedge_core::{StrategyConfig, SimRunner};
use quantedge_data::{BarStream, BarLoadConfig, LotSizes, ExpiryCalendar};
use quantedge_metrics::MetricsEngine;
use std::time::Instant;

#[derive(Parser)]
#[command(name = "backtest", about = "QuantEdge Backtester CLI")]
struct Args {
    /// Path to strategy TOML file
    #[arg(short, long)]
    strategy: String,

    /// Path to Parquet data directory
    #[arg(short, long, default_value = "data/parquet")]
    data_dir: String,

    /// Path to lot sizes config
    #[arg(long, default_value = "config/lot_sizes.toml")]
    lot_config: String,

    /// Output format: table or json
    #[arg(long, default_value = "table")]
    output: String,
}

fn main() -> Result<(), Box<dyn std::error::Error>> {
    let args = Args::parse();
    let start = Instant::now();

    // 1. Parse strategy
    let config = StrategyConfig::from_toml(&args.strategy)?;
    println!("Strategy: {}", config.name);
    println!("Underlying: {}", config.underlying);

    // 2. Load bars
    let bar_config = BarLoadConfig {
        symbol: config.underlying.clone(),
        expiry_type: config.expiry_filter.clone(),
        data_dir: args.data_dir.clone(),
        ..Default::default()
    };
    let bars = BarStream::load(&bar_config)?;
    println!("Loaded {} bars in {:?}", bars.len(), start.elapsed());

    // 3. Get lot size
    let lots = LotSizes::from_toml(&args.lot_config)?;
    let lot_size = lots.get(&config.underlying, bars.first()
        .map_or(chrono::NaiveDate::from_ymd_opt(2024,1,1).unwrap(), |b| b.date))
        .unwrap_or(15);

    // 4. Run simulation
    let sim_start = Instant::now();
    let result = SimRunner::run(&config, &bars, lot_size as u32);
    let sim_elapsed = sim_start.elapsed();

    // 5. Compute metrics
    let start_date = bars.first().map_or(/* default */, |b| b.date);
    let end_date = bars.last().map_or(/* default */, |b| b.date);
    let metrics = MetricsEngine::compute(&result.trades, &result.snapshots,
        config.capital, start_date, end_date);

    // 6. Print results
    println!("\n━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━");
    println!(" QuantEdge Backtest Results");
    println!("━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━");
    println!(" Total PnL (net):   ₹{:.2}", metrics.total_pnl_net);
    println!(" CAGR:              {:.2}%", metrics.cagr * 100.0);
    println!(" Win Rate:          {:.1}%", metrics.win_rate_pct);
    println!(" Profit Factor:     {:.2}", metrics.profit_factor);
    println!(" Sharpe Ratio:      {:.2}", metrics.sharpe_ratio);
    println!(" Max Drawdown:      ₹{:.2} ({:.1}%)", metrics.max_drawdown_inr, metrics.max_drawdown_pct);
    println!(" Total Trades:      {}", metrics.total_trades);
    println!(" Expectancy:        ₹{:.2}", metrics.expectancy);
    println!("━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━");
    println!(" Simulation time:   {:?}", sim_elapsed);
    println!(" Total time:        {:?}", start.elapsed());

    if args.output == "json" {
        println!("{}", serde_json::to_string_pretty(&metrics)?);
    }

    Ok(())
}
```

```rust
/// Tests: TOML parse → SimRunner → MetricsEngine → verify results

#[test]
fn test_full_pipeline_short_call() {
    // 1. Parse strategy from embedded TOML string
    let config = StrategyConfig::from_toml_str(EXAMPLE_TOML).unwrap();
    assert_eq!(config.name, "Short ATM Call Weekly");

    // 2. Generate 5 days of synthetic bars
    // Each day: 09:15-15:30, 1-min bars, ATM CE/PE
    // Day 1: CE entry=200, exits at 185 (profit, time exit)
    // Day 2: CE entry=210, hits SL at 420 (100% SL)
    // Day 3: CE entry=190, exits at 150 (profit, time exit)
    // Day 4: CE entry=205, exits at 195 (profit, time exit)
    // Day 5: CE entry=215, hits SL at 430 (100% SL)
    let bars = generate_synthetic_bars(5);

    // 3. Run simulation
    let result = SimRunner::run(&config, &bars, 15);
    assert_eq!(result.trades.len(), 5);

    // 4. Verify trade PnL
    assert!(result.trades[0].pnl_gross > 0.0); // Day 1 profit
    assert!(result.trades[1].pnl_gross < 0.0); // Day 2 SL loss

    // 5. Compute metrics
    let metrics = MetricsEngine::compute(...);
    assert_eq!(metrics.total_trades, 5);
    assert!(metrics.win_rate_pct > 0.0);
    assert!(metrics.profit_factor > 0.0);

    // 6. Verify brokerage deducted
    assert!(result.trades.iter().all(|t| t.brokerage > 0.0));
    assert!(result.trades.iter().all(|t| t.pnl_net < t.pnl_gross));
}

#[test]
fn test_no_data_produces_zero_trades() {
    let config = StrategyConfig::from_toml_str(EXAMPLE_TOML).unwrap();
    let result = SimRunner::run(&config, &[], 15);
    assert_eq!(result.trades.len(), 0);
}

#[test]
fn test_snapshots_captured_per_bar() {
    let config = StrategyConfig::from_toml_str(EXAMPLE_TOML).unwrap();
    let bars = generate_synthetic_bars(1);
    let result = SimRunner::run(&config, &bars, 15);
    assert!(!result.snapshots.is_empty());
    // First snapshot equity should equal capital
    assert!((result.snapshots[0].equity - config.capital).abs() < 1.0);
}
```
