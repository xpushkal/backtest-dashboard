## Objective

Upgrade the simulation engine from single-leg to N-leg. Create a `Strategy` struct
that wraps `StrategyConfig` with runtime state, and `CombinedSlMonitor` that tracks
the aggregate PnL across all legs for overall/combined SL and target checks.

## must_haves

- Strategy struct with N legs runtime state management
- CombinedSlMonitor tracking sum-of-leg-PnLs against overall thresholds
- SimRunner handles multi-leg entry (opens all legs at same timestamp)
- SimRunner handles multi-leg exit (closes all legs when any exit fires)
- Multi-leg ClosedTrade records one trade per leg with shared exit reason
- PositionSnapshot includes per-leg unrealized PnL breakdown

use crate::config::{ExitReason, OverallConfig, StrategyConfig};
use crate::position::Position;

/// Combined SL/target monitor for overall (multi-leg aggregate) exits.
pub struct CombinedSlMonitor {
    pub overall: OverallConfig,
}

impl CombinedSlMonitor {
    pub fn new(overall: &OverallConfig) -> Self {
        Self { overall: overall.clone() }
    }

    /// Check if combined (overall) SL is triggered.
    /// Uses sum of all leg PnLs vs the overall threshold.
    pub fn check_overall_sl(&self, total_pnl: f64, total_entry_premium: f64) -> Option<ExitReason> {
        if !self.overall.overall_sl_enabled {
            return None;
        }
        match self.overall.overall_sl_type {
            crate::config::SlType::PercentOfPremium => {
                let loss_pct = if total_entry_premium > 0.0 {
                    (-total_pnl / total_entry_premium) * 100.0
                } else { 0.0 };
                if loss_pct >= self.overall.overall_sl_value {
                    Some(ExitReason::StopLoss)
                } else { None }
            }
            crate::config::SlType::CombinedPremium => {
                if -total_pnl >= self.overall.overall_sl_value {
                    Some(ExitReason::StopLoss)
                } else { None }
            }
            crate::config::SlType::Points => {
                if -total_pnl >= self.overall.overall_sl_value {
                    Some(ExitReason::StopLoss)
                } else { None }
            }
            _ => None,
        }
    }

    /// Check if combined (overall) target is hit.
    pub fn check_overall_target(&self, total_pnl: f64, total_entry_premium: f64) -> Option<ExitReason> {
        if !self.overall.overall_target_enabled {
            return None;
        }
        match self.overall.overall_target_type {
            crate::config::SlType::PercentOfPremium => {
                let profit_pct = if total_entry_premium > 0.0 {
                    (total_pnl / total_entry_premium) * 100.0
                } else { 0.0 };
                if profit_pct >= self.overall.overall_target_value {
                    Some(ExitReason::Target)
                } else { None }
            }
            crate::config::SlType::Points => {
                if total_pnl >= self.overall.overall_target_value {
                    Some(ExitReason::Target)
                } else { None }
            }
            _ => None,
        }
    }
}
```

   fn check_exits(config: &StrategyConfig, pos: &Position, time: NaiveTime) -> Option<ExitReason> {
       // Priority 1: Per-leg SL
       for leg in &pos.legs { if let Some(r) = leg.check_sl() { return Some(r); } }

       // Priority 2: Combined / Overall SL  ← NEW
       let monitor = CombinedSlMonitor::new(&config.overall);
       let total_pnl = pos.total_unrealized_pnl();
       let total_entry_premium: f64 = pos.legs.iter()
           .map(|l| l.entry_price * l.quantity())
           .sum();
       if let Some(r) = monitor.check_overall_sl(total_pnl, total_entry_premium) {
           return Some(r);
       }

       // Priority 3: Per-leg target
       for leg in &pos.legs { if let Some(r) = leg.check_target() { return Some(r); } }

       // Priority 4: Overall target  ← NEW
       if let Some(r) = monitor.check_overall_target(total_pnl, total_entry_premium) {
           return Some(r);
       }

       // Priority 5: Time exit
       if time >= config.exit_time() { return Some(ExitReason::TimeExit); }
       None
   }
   ```
