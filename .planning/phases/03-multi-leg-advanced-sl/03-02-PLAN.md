use crate::config::SlType;

/// Context for SL evaluation.
pub struct SlContext {
    pub entry_price: f64,
    pub current_price: f64,
    pub entry_spot: f64,
    pub current_spot: f64,
    pub quantity: f64,       // lots * lot_size
    pub lots: u32,
    pub lot_size: u32,
    pub direction: f64,      // +1 buy, -1 sell
    pub unrealized_pnl: f64,
}

/// Check if a stop-loss is triggered given the SL type, value, and market context.
pub fn is_sl_triggered(sl_type: &SlType, sl_value: f64, ctx: &SlContext) -> bool {
    match sl_type {
        SlType::None => false,

        SlType::Points => {
            // Price moved X points against position
            let price_move = (ctx.current_price - ctx.entry_price) * -ctx.direction;
            price_move >= sl_value
        }

        SlType::PercentOfPremium => {
            // Loss as % of entry premium * quantity
            let loss_pct = if ctx.entry_price > 0.0 {
                (-ctx.unrealized_pnl / (ctx.entry_price * ctx.quantity)) * 100.0
            } else { 0.0 };
            loss_pct >= sl_value
        }

        SlType::PercentOfMargin => {
            // Simplified SPAN margin: max(3 × premium × lots, factor × spot × lots × 0.12)
            // factor depends on index (BankNifty=0.25, Nifty=0.15, Sensex=0.20)
            // Use 0.20 as default for now
            let premium_margin = 3.0 * ctx.entry_price * ctx.lots as f64 * ctx.lot_size as f64;
            let span_margin = 0.20 * ctx.entry_spot * ctx.lots as f64 * ctx.lot_size as f64 * 0.12;
            let margin = premium_margin.max(span_margin);
            let loss_pct = if margin > 0.0 {
                (-ctx.unrealized_pnl / margin) * 100.0
            } else { 0.0 };
            loss_pct >= sl_value
        }

        SlType::IndexPoints => {
            // Spot moved X index points against position
            let spot_move = (ctx.current_spot - ctx.entry_spot) * -ctx.direction;
            spot_move >= sl_value
        }

        SlType::DeltaBreach => {
            // Stub for Phase 3 — requires Greeks engine (Phase 5)
            // Will check if leg delta crosses threshold
            false
        }

        SlType::CombinedPremium => {
            // This is checked at the position/strategy level, not per-leg
            // CombinedSlMonitor handles this in strategy.rs
            false
        }
    }
}

/// Check if a target is triggered.
pub fn is_target_triggered(target_type: &SlType, target_value: f64, ctx: &SlContext) -> bool {
    match target_type {
        SlType::PercentOfPremium => {
            let profit_pct = if ctx.entry_price > 0.0 {
                (ctx.unrealized_pnl / (ctx.entry_price * ctx.quantity)) * 100.0
            } else { 0.0 };
            profit_pct >= target_value
        }
        SlType::Points => {
            let price_move = (ctx.current_price - ctx.entry_price) * ctx.direction;
            price_move >= target_value
        }
        _ => false,
    }
}

use crate::sl_types::{is_sl_triggered, is_target_triggered, SlContext};

impl Leg {
    fn make_sl_context(&self) -> SlContext {
        SlContext {
            entry_price: self.entry_price,
            current_price: self.current_price,
            entry_spot: self.entry_spot,
            current_spot: self.current_spot,
            quantity: self.quantity(),
            lots: self.lots,
            lot_size: self.lot_size,
            direction: self.direction(),
            unrealized_pnl: self.unrealized_pnl,
        }
    }

    pub fn check_sl(&self) -> Option<ExitReason> {
        if let SlState::Triggered { reason } = &self.sl_state {
            return Some(*reason);
        }
        if !self.config.stop_loss_enabled { return None; }
        let ctx = self.make_sl_context();
        if is_sl_triggered(&self.config.stop_loss_type, self.config.stop_loss_value, &ctx) {
            Some(ExitReason::StopLoss)
        } else { None }
    }

    pub fn check_target(&self) -> Option<ExitReason> {
        if !self.config.target_profit_enabled { return None; }
        let ctx = self.make_sl_context();
        if is_target_triggered(&self.config.target_profit_type, self.config.target_profit_value, &ctx) {
            Some(ExitReason::Target)
        } else { None }
    }
}
