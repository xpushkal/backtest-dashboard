## Objective

Add Delta, Premium, and Percent-OTM strike selection modes. Phase 2 only
had ATM offset. These modes select strikes based on option premium, option
delta (stubbed — requires Greeks), or percentage distance from spot.

## must_haves

- Premium mode: select strike where option premium is closest to target premium
- Percent-OTM mode: select strike at X% distance from ATM spot
- Delta mode: stubbed (selects ATM+0 as fallback; live in Phase 5 with Greeks engine)
- StrikeSelector::select handles all 4 modes
- All modes produce a valid (index, price, spot) tuple or None

```rust
StrikeMode::Premium => {
    // Find strike where close price is closest to leg.premium_target
    let target = leg.premium_target.unwrap_or(200.0);
    let mut best: Option<(usize, f64, f64)> = None;
    let mut best_diff = f64::MAX;
    for i in 0..option_types.len() {
        if option_types[i] == target_type {
            let diff = (closes[i] - target).abs();
            if diff < best_diff {
                best_diff = diff;
                best = Some((i, closes[i], spots[i]));
            }
        }
    }
    best
}

StrikeMode::PercentOtm => {
    // Find strike at X% OTM from spot
    // OTM% = |strike - spot| / spot * 100
    // For CE: OTM if strike > spot. For PE: OTM if strike < spot.
    // Need strike prices in the bar data (strike_offset * step_size + ATM)
    // Use strike_offset as proxy: select offset closest to desired %
    let target_pct = leg.strike_offset as f64; // reuse strike_offset as % target
    // In Phase 3, we use approximate matching via strike_offset
    // Full implementation requires actual strike prices (Phase 5)
    let mut best: Option<(usize, f64, f64)> = None;
    let mut best_diff = i32::MAX;
    for i in 0..option_types.len() {
        if option_types[i] == target_type {
            let diff = (strike_offsets[i] - leg.strike_offset).abs();
            if diff < best_diff {
                best_diff = diff;
                best = Some((i, closes[i], spots[i]));
            }
        }
    }
    best
}

StrikeMode::Delta => {
    // Stub: requires Greeks engine (Phase 5)
    // Fallback to ATM+0 for now
    for i in 0..option_types.len() {
        if option_types[i] == target_type && strike_offsets[i] == 0 {
            return Some((i, closes[i], spots[i]));
        }
    }
    None
}
```

```rust
fn test_select_premium_mode() {
    let leg = make_leg_with_mode(OptionType::CE, StrikeMode::Premium, 150.0);
    let types = vec!["CE", "CE", "CE"];
    let offsets = vec![-5, 0, 5];
    let closes = vec![300.0, 200.0, 120.0]; // 120 closest to target 150

    let (idx, price, _) = StrikeSelector::select(&leg, &types, &offsets, &closes, &spots).unwrap();
    assert_eq!(idx, 2);
    assert_eq!(price, 120.0);
```
