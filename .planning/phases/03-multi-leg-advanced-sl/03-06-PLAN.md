Expand `StrategyConfig::validate()` with additional checks:

```rust
pub fn validate(&self) -> Result<(), String> {
    // Existing checks
    if self.strategy.capital <= 0.0 {
        return Err("Capital must be positive".to_string());
    }
    if self.legs.is_empty() {
        return Err("At least one leg is required".to_string());
    }
    if self.entry_time() >= self.exit_time() {
        return Err("Entry time must be before exit time".to_string());
    }

    // New checks
    for (i, leg) in self.legs.iter().enumerate() {
        // Trailing SL validation
        if leg.trail_sl_enabled && leg.trail_sl_activate_at <= 0.0 {
            return Err(format!(
                "Leg {}: trailing SL enabled but activate_at is 0. Set trail_sl_activate_at > 0.",
                i + 1
            ));
        }

        // SL value must be positive when enabled
        if leg.stop_loss_enabled && leg.stop_loss_value <= 0.0 {
            return Err(format!(
                "Leg {}: stop loss enabled but value is 0. Set stop_loss_value > 0.",
                i + 1
            ));
        }

        // Target value must be positive when enabled
        if leg.target_profit_enabled && leg.target_profit_value <= 0.0 {
            return Err(format!(
                "Leg {}: target profit enabled but value is 0. Set target_profit_value > 0.",
                i + 1
            ));
        }

        // Lots must be positive
        if leg.lots == 0 {
            return Err(format!("Leg {}: lots must be > 0.", i + 1));
        }
    }

    // Overall SL with fewer than 2 legs is meaningless
    if self.overall.overall_sl_enabled && self.legs.len() < 2 {
        return Err("Overall SL requires at least 2 legs. Use per-leg SL for single-leg strategies.".to_string());
    }

    Ok(())
}
```
</action>

- `validate()` warns: overall SL on single-leg
- Error messages include leg index (1-based) for per-leg errors
- `cargo check -p quantedge-core` compiles
</acceptance_criteria>
